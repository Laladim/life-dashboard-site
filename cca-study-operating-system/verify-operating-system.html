CCA Operating System Verifier

This page shows the verifier script in readable form. Run it from the trainer folder with python3 verify_operating_system.py.

#!/usr/bin/env python3
"""Verify the CCA trainer operating-system artifacts.

This is a lightweight structural verifier. It does not grade Shela's knowledge;
it checks that the operating system has the required files and coverage scaffolds.
"""

from pathlib import Path
import re
import sys


ROOT = Path(__file__).resolve().parent

REQUIRED_FILES = [
    "SKILL.md",
    "CCA-Operating-System.html",
    "knowledge-base.md",
    "progress.md",
    "commitments-tracker.md",
    "guide-mastery-matrix.md",
    "daily-study-loop.md",
    "readiness-gates.md",
    "daily-review-sheet.md",
    "mastery-dashboard.md",
    "trainer-session-checklist.md",
    "mock-exam-bank.md",
    "weekly-review-protocol.md",
    "operating-system-review.md",
]

FORBIDDEN_PATTERNS = [
    r"60\s*Q",
    r"120\s*min",
    r"Launched March 12",
    r"source of truth:\s*knowledge-base",
    r"Domain 4.*tool descriptions",
]


def read(name: str) -> str:
    return (ROOT / name).read_text(encoding="utf-8")


def fail(message: str, failures: list[str]) -> None:
    failures.append(message)


def count_task_rows(matrix: str) -> int:
    return len(re.findall(r"^\| [1-5]\.\d+ \|", matrix, flags=re.MULTILINE))


def count_scenario_rows(matrix: str) -> int:
    scenario_section = matrix.split("## Scenario Mastery", 1)[-1]
    return len(re.findall(r"^\| (Customer Support Resolution Agent|Code Generation with Claude Code|Multi-Agent Research System|Developer Productivity with Claude|Claude Code for Continuous Integration|Structured Data Extraction) \|", scenario_section, flags=re.MULTILINE))


def count_mock_questions(bank: str) -> int:
    return len(re.findall(r"^### Q\d+", bank, flags=re.MULTILINE))


def parse_matrix_rows(matrix: str) -> list[dict[str, str]]:
    rows: list[dict[str, str]] = []
    for line in matrix.splitlines():
        if not re.match(r"^\| [1-5]\.\d+ \|", line):
            continue
        parts = [part.strip() for part in line.strip("|").split("|")]
        if len(parts) < 6:
            continue
        rows.append(
            {
                "id": parts[0],
                "domain": parts[1],
                "task": parts[2],
                "meaning": parts[3],
                "status": parts[4],
                "evidence": parts[5],
            }
        )
    return rows


def parse_scenario_rows(matrix: str) -> list[dict[str, str]]:
    rows: list[dict[str, str]] = []
    scenario_section = matrix.split("## Scenario Mastery", 1)[-1]
    known = {
        "Customer Support Resolution Agent",
        "Code Generation with Claude Code",
        "Multi-Agent Research System",
        "Developer Productivity with Claude",
        "Claude Code for Continuous Integration",
        "Structured Data Extraction",
    }
    for line in scenario_section.splitlines():
        if not line.startswith("| "):
            continue
        parts = [part.strip() for part in line.strip("|").split("|")]
        if len(parts) < 4 or parts[0] not in known:
            continue
        rows.append(
            {
                "scenario": parts[0],
                "task": parts[1],
                "status": parts[2],
                "evidence": parts[3],
            }
        )
    return rows


def status_counts(rows: list[dict[str, str]]) -> dict[str, int]:
    counts: dict[str, int] = {}
    for row in rows:
        counts[row["status"]] = counts.get(row["status"], 0) + 1
    return counts


def write_dashboard(task_rows: list[dict[str, str]], scenario_rows: list[dict[str, str]]) -> None:
    counts = status_counts(task_rows)
    scenario_counts = status_counts(scenario_rows)
    practiced_or_passed = sum(1 for row in task_rows if row["status"] in {"Practiced", "Passed"})
    passed = sum(1 for row in task_rows if row["status"] == "Passed")
    needs_review = [row for row in task_rows if row["status"] == "Needs review"]
    not_started = [row for row in task_rows if row["status"] == "Not started"]

    lines = [
        "# CCA Mastery Dashboard",
        "",
        "Generated by `verify_operating_system.py` from `guide-mastery-matrix.md`.",
        "",
        "## Coverage",
        "",
        f"- Guide task rows: {len(task_rows)}",
        f"- Practiced or passed: {practiced_or_passed} / {len(task_rows)}",
        f"- Passed: {passed} / {len(task_rows)}",
        f"- Scenario rows: {len(scenario_rows)}",
        "",
        "## Task Status Counts",
        "",
    ]
    for status in ["Not started", "Explained", "Practiced", "Passed", "Needs review"]:
        lines.append(f"- {status}: {counts.get(status, 0)}")

    lines.extend(["", "## Scenario Status Counts", ""])
    for status in ["Not started", "Explained", "Practiced", "Passed", "Needs review"]:
        lines.append(f"- {status}: {scenario_counts.get(status, 0)}")

    lines.extend(["", "## Highest Priority Needs Review", ""])
    if needs_review:
        for row in needs_review[:8]:
            lines.append(f"- {row['id']} {row['task']} ({row['domain']})")
    else:
        lines.append("- None marked `Needs review`.")

    lines.extend(["", "## Next Not Started Rows", ""])
    if not_started:
        for row in not_started[:8]:
            lines.append(f"- {row['id']} {row['task']} ({row['domain']})")
    else:
        lines.append("- No task rows are `Not started`.")

    lines.extend(
        [
            "",
            "## Readiness Snapshot",
            "",
            "- Gate 1 requires all 30 rows at least `Practiced`, 25 rows `Passed`, no Domain 1 row below `Practiced`, and no stale `Needs review` row.",
            "- This dashboard is structural. It does not replace graded drills or Shela's teach-back evidence.",
            "",
        ]
    )

    (ROOT / "mastery-dashboard.md").write_text("\n".join(lines), encoding="utf-8")


def main() -> int:
    failures: list[str] = []

    for filename in REQUIRED_FILES:
        if not (ROOT / filename).exists():
            fail(f"Missing required file: {filename}", failures)

    if failures:
        print("CCA OS VERIFY: FAIL")
        for item in failures:
            print(f"- {item}")
        return 1

    combined = "\n".join(read(name) for name in REQUIRED_FILES if name.endswith(".md"))
    for pattern in FORBIDDEN_PATTERNS:
        if re.search(pattern, combined, flags=re.IGNORECASE):
            fail(f"Forbidden/stale pattern present: {pattern}", failures)

    skill = read("SKILL.md")
    for required_ref in [
        "guide-mastery-matrix.md",
        "daily-study-loop.md",
        "readiness-gates.md",
        "trainer-session-checklist.md",
    ]:
        if required_ref not in skill:
            fail(f"SKILL.md does not require/reference {required_ref}", failures)

    matrix = read("guide-mastery-matrix.md")
    task_count = count_task_rows(matrix)
    if task_count != 30:
        fail(f"Expected 30 guide task rows, found {task_count}", failures)

    scenario_count = count_scenario_rows(matrix)
    if scenario_count != 6:
        fail(f"Expected 6 scenario rows, found {scenario_count}", failures)

    task_rows = parse_matrix_rows(matrix)
    scenario_rows = parse_scenario_rows(matrix)
    write_dashboard(task_rows, scenario_rows)

    readiness = read("readiness-gates.md")
    for gate in range(1, 8):
        if f"Gate {gate}" not in readiness:
            fail(f"Missing readiness Gate {gate}", failures)

    bank = read("mock-exam-bank.md")
    mock_count = count_mock_questions(bank)
    if mock_count != 30:
        fail(f"Expected 30 mock questions, found {mock_count}", failures)

    if "practice questions, not official exam questions" not in bank:
        fail("Mock bank does not clearly label questions as practice", failures)

    page = read("CCA-Operating-System.html")
    for required_text in [
        "How the CCA study operating system works",
        "The trainer loop",
        "Current mastery snapshot",
        "Daily study workflow",
        "Readiness gates",
        "Operating files",
    ]:
        if required_text not in page:
            fail(f"Operating-system HTML missing section text: {required_text}", failures)

    if failures:
        print("CCA OS VERIFY: FAIL")
        for item in failures:
            print(f"- {item}")
        return 1

    print("CCA OS VERIFY: PASS")
    print(f"- Required files: {len(REQUIRED_FILES)}")
    print(f"- Guide task rows: {task_count}")
    print(f"- Scenario rows: {scenario_count}")
    print(f"- Mock questions: {mock_count}")
    print("- Readiness gates: 7")
    print("- Mastery dashboard: refreshed")
    print("- HTML operating page: present")
    return 0


if __name__ == "__main__":
    sys.exit(main())
